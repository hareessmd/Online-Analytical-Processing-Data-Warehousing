CREATE TABLE ADMIN.SHIPPING 
    ( 
     SHIPPINGID       NUMBER , 
     SHIPPINGDATE     DATE , 
     ORDERID          NUMBER , 
     ORDERDATE        DATE , 
     MODE_OF_SHIPPING VARCHAR2 (20) , 
     DELIVEREDDATE    DATE , 
     SHIPPING_ADDRESS VARCHAR2 (20) , 
     "Zip Code"       NUMBER , 
     CUSTOMERID       NUMBER , 
     SERIALNUMBER     NUMBER GENERATED ALWAYS AS IDENTITY 
        ( START WITH 1 CACHE 20 )  NOT NULL 
    ) 
    TABLESPACE DATA 
    LOGGING 
;


CREATE UNIQUE INDEX ADMIN.SHIPPING_PK ON ADMIN.SHIPPING 
    ( 
     SHIPPINGID ASC 
    ) 
    TABLESPACE DATA 
    LOGGING 
;

ALTER TABLE ADMIN.SHIPPING 
    ADD CONSTRAINT SHIPPING_PK PRIMARY KEY ( SHIPPINGID ) 
    USING INDEX ADMIN.SHIPPING_PK ;

ALTER TABLE ADMIN.SHIPPING 
    ADD CONSTRAINT SHIP_TO_CUST FOREIGN KEY 
    ( 
     CUSTOMERID
    ) 
    REFERENCES ADMIN.CUSTOMER ( CUSTOMERID ) 
    ON DELETE CASCADE 
    NOT DEFERRABLE 
;

ALTER TABLE ADMIN.SHIPPING 
    ADD CONSTRAINT SHIP_TO_ORD FOREIGN KEY 
    ( 
     ORDERID
    ) 
    REFERENCES ADMIN."ORDER" ( ORDERID ) 
    ON DELETE CASCADE 
    NOT DEFERRABLE 
;
