CREATE TABLE ADMIN.PRODUCT 
    ( 
     PRODUCTID    NUMBER , 
     PRODUCT_NAME VARCHAR2 (128) , 
     COST         NUMBER , 
     QUANTITY     NUMBER , 
     STOREID      NUMBER , 
     SERIALNUMBER NUMBER GENERATED ALWAYS AS IDENTITY 
        ( START WITH 1 CACHE 20 )  NOT NULL 
    ) 
    TABLESPACE DATA 
    LOGGING 
;


CREATE UNIQUE INDEX ADMIN.PRODUCT_PK ON ADMIN.PRODUCT 
    ( 
     PRODUCTID ASC 
    ) 
    TABLESPACE DATA 
    LOGGING 
;

ALTER TABLE ADMIN.PRODUCT 
    ADD CONSTRAINT PRODUCT_PK PRIMARY KEY ( PRODUCTID ) 
    USING INDEX ADMIN.PRODUCT_PK ;

ALTER TABLE ADMIN.PRODUCT 
    ADD CONSTRAINT PRO_TO_STORE FOREIGN KEY 
    ( 
     STOREID
    ) 
    REFERENCES ADMIN.STORES ( STOREID ) 
    ON DELETE CASCADE 
    NOT DEFERRABLE 
;
