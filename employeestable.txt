CREATE TABLE ADMIN.EMPLOYEES 
    ( 
     EMPLOYEEID   NUMBER , 
     FNAME        VARCHAR2 (20) , 
     LNAME        VARCHAR2 (20) , 
     STOREID      NUMBER , 
     ROLE         VARCHAR2 (20) , 
     ADDRESS      VARCHAR2 (20) , 
     SERIALNUMBER NUMBER GENERATED ALWAYS AS IDENTITY 
        ( START WITH 1 CACHE 20 )  NOT NULL 
    ) 
    TABLESPACE DATA 
    LOGGING 
;


CREATE UNIQUE INDEX ADMIN.EMPLOYEES_PK ON ADMIN.EMPLOYEES 
    ( 
     EMPLOYEEID ASC 
    ) 
    TABLESPACE DATA 
    LOGGING 
;

ALTER TABLE ADMIN.EMPLOYEES 
    ADD CONSTRAINT EMPLOYEES_PK PRIMARY KEY ( EMPLOYEEID ) 
    USING INDEX ADMIN.EMPLOYEES_PK ;

ALTER TABLE ADMIN.EMPLOYEES 
    ADD CONSTRAINT EMP_TO_STORE FOREIGN KEY 
    ( 
     STOREID
    ) 
    REFERENCES ADMIN.STORES ( STOREID ) 
    ON DELETE CASCADE 
    NOT DEFERRABLE 
;
