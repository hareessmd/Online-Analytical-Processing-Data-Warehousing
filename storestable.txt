CREATE TABLE ADMIN.STORES 
    ( 
     STOREID      NUMBER , 
     STORENAME    VARCHAR2 (20) , 
     STORETYPE    VARCHAR2 (20) , 
     ADDRESS      VARCHAR2 (20) , 
     CITY         VARCHAR2 (20) , 
     STATE        VARCHAR2 (20) , 
     SERIALNUMBER NUMBER GENERATED ALWAYS AS IDENTITY 
        ( START WITH 1 CACHE 20 )  NOT NULL 
    ) 
    TABLESPACE DATA 
    LOGGING 
;


CREATE UNIQUE INDEX ADMIN.STORES_PK ON ADMIN.STORES 
    ( 
     STOREID ASC 
    ) 
    TABLESPACE DATA 
    LOGGING 
;

ALTER TABLE ADMIN.STORES 
    ADD CONSTRAINT STORES_PK PRIMARY KEY ( STOREID ) 
    USING INDEX ADMIN.STORES_PK ;
